Data Repositories (Simulated Database) self.active_currency = "INR" # Multi-Currency Architecture self.banks_db = [ {"name": "Kotak Mahindra Bank (7587)", "type": "Current"}, {"name": "State Bank of India (1102)", "type": "CC"} ] # Default Mode self.create_header() self.create_main_layout() self.update_purpose_dropdown() def create_header(self): # Top Global Vault Header header_frame = tk.Frame(self.root, bg="#1C2541", height=70) header_frame.pack(fill="x", side="top") title_label = tk.Label(header_frame, text="TRANSACTION MATRIX CORE", font=("Consolas", 16, "bold"), fg="#4EA8DE", bg="#1C2541") title_label.pack(side="left", padx=20, pady=15) # Live Multi-Currency Balances balance_frame = tk.Frame(header_frame, bg="#1C2541") balance_frame.pack(side="right", padx=20) inr_bal = tk.Label(balance_frame, text="GLOBAL VAULT (INR): ₹ 48,50,000", font=("Consolas", 11, "bold"), fg="#52B788", bg="#1C2541") inr_bal.pack(side="left", padx=10) usd_bal = tk.Label(balance_frame, text="GLOBAL VAULT (USD): $ 25,400", font=("Consolas", 11, "bold"), fg="#4EA8DE", bg="#1C2541") usd_bal.pack(side="left", padx=10) def create_main_layout(self): # Main split: Left Entry / Right Database Panel self.main_container = tk.Frame(self.root, bg="#0B132B") self.main_container.pack(fill="both", expand=True, padx=15, pady=15) self.left_panel = tk.Frame(self.main_container, bg="#111D4A", bd=2, relief="groove") self.left_panel.pack(side="left", fill="both", expand=True, padx=(0, 10)) self.right_panel = tk.Frame(self.main_container, bg="#1C2541", bd=2, relief="groove", width=450) self.right_panel.pack(side="right", fill="both", padx=(10, 0)) self.right_panel.pack_propagate(False) self.setup_left_entry_form() self.setup_right_database_panel() def setup_left_entry_form(self): # Credit/Debit Switch Buttons btn_frame = tk.Frame(self.left_panel, bg="#111D4A") btn_frame.pack(fill="x", pady=15, padx=20) self.credit_btn = tk.Button(btn_frame, text="CREDIT", bg="#1C2541", fg="#FFFFFF", font=("Arial", 11, "bold"), command=lambda: self.switch_mode("CREDIT"), width=15) self.credit_btn.pack(side="left", padx=5) self.debit_btn = tk.Button(btn_frame, text="DEBIT", bg="#E63946", fg="#FFFFFF", font=("Arial", 11, "bold"), command=lambda: self.switch_mode("DEBIT"), width=15) self.debit_btn.pack(side="left", padx=5) # Form Fields form_frame = tk.Frame(self.left_panel, bg="#111D4A") form_frame.pack(fill="both", expand=True, padx=20) # Bank Selector tk.Label(form_frame, text="SELECT MATRIX BANK:", bg="#111D4A", fg="#4EA8DE").grid(row=0, column=0, sticky="w", pady=5) self.bank_var = tk.StringVar() self.bank_cb = ttk.Combobox(form_frame, textvariable=self.bank_var, state="readonly", width=35) self.bank_cb['values'] = [b["name"] for b in self.banks_db] self.bank_cb.current(0) self.bank_cb.grid(row=1, column=0, sticky="w", pady=5) # Transaction Method tk.Label(form_frame, text="TRANSACTION METHOD:", bg="#111D4A", fg="#4EA8DE").grid(row=2, column=0, sticky="w", pady=5) self.method_var = tk.StringVar() self.method_cb = ttk.Combobox(form_frame, textvariable=self.method_var, state="readonly", width=35) self.method_cb['values'] = ["ONLINE / IMPS", "RTGS / NEFT", "CASH / ANGADIA REPO"] self.method_cb.current(0) self.method_cb.grid(row=3, column=0, sticky="w", pady=5) # Particular / Party Name tk.Label(form_frame, text="PARTICULAR / PARTY NAME:", bg="#111D4A", fg="#4EA8DE").grid(row=4, column=0, sticky="w", pady=5) self.party_entry = tk.Entry(form_frame, width=38, font=("Arial", 10), bg="#1C2541", fg="#FFFFFF", insertbackground="white") self.party_entry.insert(0, "Enter Entity Name") self.party_entry.grid(row=5, column=0, sticky="w", pady=5) # Transaction Purpose tk.Label(form_frame, text="TRANSACTION PURPOSE:", bg="#111D4A", fg="#4EA8DE").grid(row=0, column=1, sticky="w", pady=5, padx=20) self.purpose_var = tk.StringVar() self.purpose_cb = ttk.Combobox(form_frame, textvariable=self.purpose_var, state="readonly", width=35) self.purpose_cb.grid(row=1, column=1, sticky="w", pady=5, padx=20) # Value Amount & Currency Selector tk.Label(form_frame, text="VALUE AMOUNT & CURRENCY:", bg="#111D4A", fg="#4EA8DE").grid(row=2, column=1, sticky="w", pady=5, padx=20) amt_curr_frame = tk.Frame(form_frame, bg="#111D4A") amt_curr_frame.grid(row=3, column=1, sticky="w", pady=5, padx=20) self.amount_entry = tk.Entry(amt_curr_frame, width=22, font=("Arial", 10), bg="#1C2541", fg="#FFFFFF", insertbackground="white") self.amount_entry.insert(0, "0.00") self.amount_entry.pack(side="left") self.curr_var = tk.StringVar(value="INR") self.curr_cb = ttk.Combobox(amt_curr_frame, textvariable=self.curr_var, values=["INR", "USD"], state="readonly", width=8) self.curr_cb.pack(side="left", padx=5) # Execution Date tk.Label(form_frame, text="EXECUTION DATE:", bg="#111D4A", fg="#4EA8DE").grid(row=4, column=1, sticky="w", pady=5, padx=20) self.date_entry = tk.Entry(form_frame, width=38, font=("Arial", 10), bg="#1C2541", fg="#FFFFFF") self.date_entry.insert(0, datetime.now().strftime("%d-%m-%Y")) self.date_entry.grid(row=5, column=1, sticky="w", pady=5, padx=20) # Action Button self.post_btn = tk.Button(self.left_panel, text="POST MATRIX ENTRY (DEBIT / OUTGOING)", bg="#E63946", fg="#FFFFFF", font=("Arial", 12, "bold"), height=2, command=self.handle_post_entry) self.post_btn.pack(fill="x", side="bottom", padx=20, pady=20) def setup_right_database_panel(self): # Top Navigation inside right panel top_nav = tk.Frame(self.right_panel, bg="#1C2541") top_nav.pack(fill="x", pady=5) bank_hier_btn = tk.Button(top_nav, text="BANK HIERARCHY", bg="#1C2541", fg="#888888", font=("Arial", 9, "bold"), bd=0, activebackground="#1C2541") bank_hier_btn.pack(side="left", padx=10) details_btn = tk.Button(top_nav, text="LIVE DETAILS PANEL", bg="#1C2541", fg="#4EA8DE", font=("Arial", 10, "underline", "bold"), bd=0, activebackground="#1C2541") details_btn.pack(side="right", padx=15) # Sub-title lbl = tk.Label(self.right_panel, text="LIVE ACCOUNT LEDGER DATABASE", font=("Arial", 11, "bold"), bg="#1C2541", fg="#FFFFFF") lbl.pack(anchor="w", padx=15, pady=5) # Intelligent Notification / Review Display Box self.review_box = tk.Text(self.right_panel, bg="#0B132B", fg="#52B788", font=("Consolas", 10), bd=1, relief="solid", highlightthickness=0, padx=10, pady=10) self.review_box.pack(fill="both", expand=True, padx=15, pady=(5, 10)) # Display simulated Bank API structure & live log instantly self.refresh_live_details_view("[SYSTEM STATUS]: Active\n[API NODE]: Waiting for secure bank response...") def refresh_live_details_view(self, message): self.review_box.config(state="normal") self.review_box.delete("1.0", tk.END) self.review_box.insert(tk.END, message) self.review_box.config(state="disabled") def switch_mode(self, mode): self.current_mode = mode if mode == "CREDIT": self.credit_btn.config(bg="#52B788") self.debit_btn.config(bg="#1C2541") self.post_btn.config(text="POST MATRIX ENTRY (CREDIT / INCOMING)", bg="#52B788") else: self.credit_btn.config(bg="#1C2541") self.debit_btn.config(bg="#E63946") self.post_btn.config(text="POST MATRIX ENTRY (DEBIT / OUTGOING)", bg="#E63946") self.update_purpose_dropdown() def update_purpose_dropdown(self): if self.current_mode == "CREDIT": self.purpose_cb['values'] = self.credit_purposes else: self.purpose_cb['values'] = self.debit_purposes self.purpose_cb.current(0) def handle_post_entry(self): # Fetch data input dynamically party = self.party_entry.get() amount = self.amount_entry.get() purpose = self.purpose_var.get() bank = self.bank_var.get() currency = self.curr_var.get() # Basic Validation if party == "Enter Entity Name" or not party.strip(): messagebox.showwarning("Validation Error", "Please provide a valid Party or Entity Name.") return # Simulate Core Engine Parsing & Real-time Live Log Feedback inside Right Box log_color = "🟢 CREDIT" if self.current_mode == "CREDIT" else "🔴 DEBIT" symbol = "$" if currency == "USD" else "₹" # Fetch Account Type from simulated DB structure to display automatically inside log box acc_type = next((b["type"] for b in self.banks_db if b["name"] == bank), "Current") success_log = ( f"=========================================\n" f" TRANSACTION VERIFICATION SUCCESSFUL\n" f"=========================================\n" f"• Status: Approved & Logged\n" f"• Action: {log_color}\n" f"• Bank Node: {bank}\n" f"• Account Type: {acc_type} Account [API Auto-Fetched]\n" f"• Entity Target: {party}\n" f"• Purpose Category: {purpose}\n" f"• Final Value: {symbol} {amount} ({currency})\n" f"• Timestamp: {datetime.now().strftime('%Y-%m-%d %H:%M:%S')}\n" f"-----------------------------------------\n" f"[SYSTEM]: Core Vault Balance adjusted successfully." ) self.refresh_live_details_view(success_log) messagebox.showinfo("Matrix Success", f"Transaction entry posted successfully under {purpose}!") if __name__ == "__main__": root = tk.Tk() app = DiamondMatrixOS(root) root.mainloop()
